# SET ENV
. ./exports.sh

# To start Mysql
docker run -p 3306:3306 \
--name serverName \
-e MYSQL_ROOT_PASSWORD=password \
-e MYSQL_DATABASE=dbName \
-e MYSQL_USER=user \
-e MYSQL_PASSWORD=password \
-d mysql

# Start PHPMYADMIN
docker run --name phpmyadmin -d -e PMA_HOST=serverName -p 8081:80 phpmyadmin
docker run --name phpmyadmin -d --link serverName:dbName -e MYSQL_ROOT_PASSWORD=password -p 8081:80 phpmyadmin


# Start the app
java -jar build/libs/bootcamp-java-mysql-project-1.0-SNAPSHOT.jar
